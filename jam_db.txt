CREATE TABLE users (
    id serial primary key,
    username text,
    curent_jam_id integer REFERENCES jams (id)
) WITH (OIDS=TRUE);

CREATE TABLE songs (
    id serial primary key,
    name  text,
    artist text,
    album text,
    album_artwork text,
    source text
) WITH (OIDS=TRUE);

CREATE TABLE jams (
    id serial primary key,
    user_id integer,
    song_id integer,
    date timestamp with time zone default now()
) WITH (OIDS=TRUE);
